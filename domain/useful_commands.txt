SELECT
    "u"."Username" AS "username",
    "s"."Name" AS "scope_name"
FROM
    "Users" AS "u"
JOIN
    "UserRoles" AS "ur" ON "u"."Id" = "ur"."UserId"
JOIN
    "Roles" AS "r" ON "ur"."RolesId" = "r"."Id"
JOIN
    "RoleScopes" AS "rs" ON "r"."Id" = "rs"."RoleId"
JOIN
    "Scopes" AS "s" ON "rs"."ScopesId" = "s"."Id"
ORDER BY
    "u"."Username",
    "s"."Name";
